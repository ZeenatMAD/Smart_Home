#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <Servo.h>

// ---------------- LCD ----------------
LiquidCrystal_I2C lcd(0x27, 16, 2);

// ---------------- Parking Servo ----------------
Servo gateServo;

// Parking IR sensors
int IR1 = 2;   // Entry IR
int IR2 = 3;   // Exit IR

// Parking slots
int slots = 4;

// Flags
bool entryInProgress = false;
bool exitInProgress  = false;

// ---------------- DUSTBIN SYSTEM ----------------
Servo lidServo;

const int trigPin = 9;
const int echoPin = 10;

long duration;
int distance;
bool isOpen = false;
unsigned long lastSeenTime = 0;

// ---------------- GATE POSITIONS ----------------
const int gateClosedUs = 2100;  // fully closed
const int gateOpenUs   = 1000;  // fully open
const int stepDelay    = 5;     // ms per step for smooth movement

// ---------------- SMOOTH GATE OPEN/CLOSE ----------------
void openGate() {
  for (int pos = gateClosedUs; pos >= gateOpenUs; pos -= 10) {
    gateServo.writeMicroseconds(pos);
    delay(stepDelay);
  }
}

void closeGate() {
  for (int pos = gateOpenUs; pos <= gateClosedUs; pos += 10) {
    gateServo.writeMicroseconds(pos);
    delay(stepDelay);
  }
}

// ---------------- ULTRASONIC FUNCTION ----------------
int getDistance() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  duration = pulseIn(echoPin, HIGH, 20000);
  int dist = duration * 0.034 / 2;

  if (dist == 0 || dist > 200)
    return 200;

  return dist;
}

// ---------------- SETUP ----------------
void setup() {
  Serial.begin(9600);

  // LCD
  lcd.begin(16, 2);
  lcd.backlight();

  // Parking servo
  gateServo.attach(6, 500, 2400);      // attach with full pulse limits
  gateServo.writeMicroseconds(gateClosedUs);

  // Dustbin servo
  lidServo.attach(7);        
  lidServo.write(0);

  // IR pins
  pinMode(IR1, INPUT);
  pinMode(IR2, INPUT);

  // Ultrasonic pins
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);

  // Startup LCD
  lcd.clear();
  lcd.print("Parking System");
  delay(1500);

  lcd.clear();
  lcd.print("Slots Left: ");
  lcd.print(slots);
}

// ---------------- LOOP ----------------
void loop() {
  int ir1 = digitalRead(IR1);
  int ir2 = digitalRead(IR2);

  // ---------------- PARKING SYSTEM ----------------
  if (!entryInProgress && !exitInProgress) {
    // CAR ENTRY
    if (ir1 == LOW && slots > 0) {
      entryInProgress = true;
      openGate();
      lcd.clear();
      lcd.print("Car Entering...");
      // don't delay here, wait for exit sensor to trigger
    }
    // CAR EXIT
    else if (ir2 == LOW && slots < 4) {
      exitInProgress = true;
      openGate();
      lcd.clear();
      lcd.print("Car Exiting...");
      // wait for entry sensor to trigger
    }
  }

  // Entry Finish
  if (entryInProgress && ir2 == LOW) {  // car has fully entered
    closeGate();
    entryInProgress = false;
    slots--;

    lcd.clear();
    lcd.print("Entry Done!");
    lcd.setCursor(0, 1);
    lcd.print("Slots Left: ");
    lcd.print(slots);
    delay(800); // small pause for user to read
  }

  // Exit Finish
  if (exitInProgress && ir1 == LOW) { // car has fully exited
    closeGate();
    exitInProgress = false;
    slots++;
    if (slots > 4) slots = 4;

    lcd.clear();
    lcd.print("Exit Done!");
    lcd.setCursor(0, 1);
    lcd.print("Slots Left: ");
    lcd.print(slots);
    delay(800); // small pause for user to read
  }

  // Refresh LCD at top line only when idle
  if (!entryInProgress && !exitInProgress) {
    lcd.setCursor(0, 0);
    lcd.print("Slots Left: ");
    lcd.print(slots);
    lcd.print("  ");
  }

  // ---------------- DUSTBIN SYSTEM ----------------
  distance = getDistance();
  Serial.print("Distance: ");
  Serial.println(distance);

  if (distance < 5) {
    if (!isOpen) {
      lidServo.write(90);
      isOpen = true;
    }
    lastSeenTime = millis();
  }

  if (isOpen && (millis() - lastSeenTime > 1200)) {
    lidServo.write(0);
    isOpen = false;
  }

  delay(50);
}
